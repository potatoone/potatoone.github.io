#include <stdio.h>

// 计算斐波那契数列
int fibonacci(int n) {
    if (n <= 1) return n;
    return fibonacci(n-1) + fibonacci(n-2);
}

int main() {
    // 打印前10项
    for (int i = 0; i < 10; i++) {
        printf("F(%d) = %d\n", i, fibonacci(i));
    }
    return 0;
}

package main

import (
	"fmt"
	"time"
)

// 定义结构体
type Person struct {
	Name string
	Age  int
}

// 结构体方法
func (p Person) Greet() string {
	return fmt.Sprintf("Hello, I'm %s, %d years old", p.Name, p.Age)

		time.Sleep(1 * time.Second)
		fmt.Println("Async task done")
	}()
	
	time.Sleep(2 * time.Second)
}

### Go语言
```
import math
from typing import List, Dict

def calculate_stats(data: List[float]) -> Dict[str, float]:
    """计算数据的统计信息"""
    if not data:
        raise ValueError("数据不能为空")
    
    return {
        "均值": sum(data) / len(data),
        "最大值": max(data),
        "最小值": min(data),
        "标准差": math.sqrt(sum((x - sum(data)/len(data))**2 for x in data) / len(data))
    }

# 测试代码
if __name__ == "__main__":
    sample_data = [12.5, 15.3, 9.8, 20.1, 17.7]
    stats = calculate_stats(sample_data)
    
    for key, value in stats.items():
        print(f"{key}: {value:.2f}")
```

### Batch
```
@echo off
REM 这是一个批处理示例
setlocal enabledelayedexpansion

REM 设置变量
set "greeting=Hello, World!"
set "count=3"

REM 循环
for /l %%i in (1,1,%count%) do (
  echo [%%i] !greeting!
  timeout /t 1 /nobreak >nul
)

REM 条件判断
if exist "test.txt" (
  echo File exists!
) else (
  echo Creating file...
  echo Sample content > test.txt
)

echo Done.
pause
```

### Sql
```
-- 创建表
CREATE TABLE users (
  id INT PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(50) NOT NULL UNIQUE,
  email VARCHAR(100) NOT NULL UNIQUE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  last_login DATETIME
);

-- 插入数据
INSERT INTO users (username, email)
VALUES 
  ('john_doe', 'john@example.com'),
  ('jane_smith', 'jane@example.com');

-- 查询数据（带条件和排序）
SELECT username, email, DATE(created_at) AS join_date
FROM users
WHERE created_at > '2023-01-01'
ORDER BY created_at DESC
LIMIT 10;

-- 更新数据
UPDATE users
SET last_login = NOW()
WHERE username = 'john_doe';
```

### DockerFile
```
# 基础镜像
FROM node:18-alpine AS builder

# 设置工作目录
WORKDIR /app

# 复制依赖文件
COPY package*.json ./

# 安装依赖
RUN npm ci

# 复制源代码
COPY . .

# 构建应用
RUN npm run build

# 生产环境镜像
FROM nginx:alpine

# 复制构建产物到Nginx
COPY --from=builder /app/dist /usr/share/nginx/html

# 暴露端口
EXPOSE 80

# 启动命令
CMD ["nginx", "-g", "daemon off;"]
```
